struct coordinateCompression {
private:
    vector<ll> values;      // unique sorted values

public:
    vector<ll> c;  // final compressed array
    coordinateCompression(vector<ll>& v) {
        values = v;
        sort(values.begin(), values.end());
        values.erase(unique(values.begin(), values.end()), values.end());
        // build mapping
        for (ll x : v) {
            int idx = lower_bound(values.begin(), values.end(), x) - values.begin();
            c.push_back(idx);
        }
    }
    // original -> compressed
    int index(ll val) {
        return lower_bound(values.begin(), values.end(), val) - values.begin();
    }
    // compressed -> original
    ll initVal(int idx) {
        return values[idx];
    }
};
